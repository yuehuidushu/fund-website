## 1. 有效的括号
```java
public class _20_isValidBrackets {
    static Map<Character, Character> bracket = new HashMap();

    static {
        bracket.put('(', ')');
        bracket.put('[', ']');
        bracket.put('{', '}');
    }

    public static boolean isValid(String s) {
        char[] arr = s.toCharArray();
        List<Character> stack = new LinkedList<>();
        int len = arr.length;

        for(int index = 0; index < len; ++index) {
            char c = arr[index];
            if (bracket.containsKey(c)) {
                stack.push(c);
            } else if (stack.isEmpty() || c != (Character)bracket.get(stack.pop())) {
                return false;
            }
        }
        return stack.isEmpty();
    }
}
```

```java
	//将栈改为数组
	public static int[] dailyTemperatures1(int[] T) {
		if(null == T || 0 == T.length) return null;
		
		int len = T.length;
		int[] result = new int[len];
		int[] stack = new int[len];
		int top = -1;
		for(int i = 0; i < len; i++) {
			while(top >= 0 && T[i] > T[stack[top]]) {
				int idx = stack[top--];
				result[idx] = i - idx;
			}
			stack[++top] = i;
		}
		return result;
	}
	
	//从后往前开始查找，效率更好
	public static int[] dailyTemperatures2(int[] T) {
		if(null == T || 0 == T.length) return null;
		
		int len = T.length;
		int[] result = new int[len];
		for (int i = len - 1; i >= 0; i--) {
			int j = i + 1;
			while (j < len) {
				if(T[j] > T[i]) {
					//如果找到就停止循环
					result[i] = j - i;
					break;
				}else if (0 == result[j]) {
					//如果没有找到，且0 == result[j]，说明ｊ后面没有更大的元素
					//则说明后面也没有比Ｔ[i]更大的元素了，因此为０．终止循环
					break;
				}else {
					//直接跳到ｊ后面比ｊ大的元素与ｉ比较，省略比ｊ小的比较
					j += result[j];
				}
			}
		}
		return result;
	}
	
	public static void main(String[] args) {
		int[] temp = {73, 74, 75, 71, 69, 72, 76, 73};
		int[] result = dailyTemperatures2(temp);
		
		for (int i = 0; i < result.length; i++) {
			System.out.println(temp[i] + "..." + result[i]);
			
		}
	}
}
```
